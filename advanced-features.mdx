## Webhooks

Webhooks allow you to integrate Karmix Alchemist with external systems by sending HTTP requests when specific events occur.

<Tabs>
  <Tab title="Setup" icon="settings">
    Configure webhooks for your projects.
    
    <Steps>
      <Step title="Create Webhook" icon="plus">
        Go to project settings and add a new webhook.
      </Step>
      <Step title="Configure Events" icon="bell">
        Choose which events should trigger the webhook.
        
        <CodeGroup tabs="Webhook Config,Event Types">
          ````json
          {
            "url": "https://api.example.com/webhook",
            "events": ["document.created", "document.updated"],
            "secret": "your-webhook-secret",
            "headers": {
              "Authorization": "Bearer token123"
            }
          }
          ````
          ````json
          {
            "document.created": "When a new document is created",
            "document.updated": "When document content changes",
            "document.published": "When document is published",
            "project.created": "When a new project is created",
            "user.invited": "When a user is invited to a project"
          }
          ````
        </CodeGroup>
      </Step>
      <Step title="Test Webhook" icon="zap">
        Send a test event to verify the integration works.
      </Step>
    </Steps>
  </Tab>
  <Tab title="Security" icon="shield">
    Secure your webhook integrations.
    
    <Callout kind="danger">Always use HTTPS URLs and verify webhook signatures to prevent unauthorized access.</Callout>
    
    <CodeGroup tabs="Signature Verification,Example Payload">
      ````javascript
      const crypto = require('crypto');
      
      function verifySignature(payload, signature, secret) {
        const expectedSignature = crypto
          .createHmac('sha256', secret)
          .update(payload)
          .digest('hex');
        return `sha256=${expectedSignature}` === signature;
      }
      ````

## API Extensions

Build custom API endpoints to extend Karmix Alchemist's functionality.

<Steps>
  <Step title="Create Extension" icon="code">
    Use our SDK to build custom endpoints.
    
    <CodeGroup tabs="Basic Extension,Advanced Extension">
      ````javascript
      const { Extension } = require('karmix-sdk');
      
      class CustomExtension extends Extension {
        registerRoutes(router) {
          router.get('/custom/stats', async (req, res) => {
            const stats = await this.getDocumentStats();
            res.json(stats);
          });
        }
        
        async getDocumentStats() {
          // Custom logic here
          return { total: 150, published: 120 };
        }
      }
      
      module.exports = CustomExtension;
      ````
      ````javascript
      const { Extension, Database } = require('karmix-sdk');
      
      class AnalyticsExtension extends Extension {
        constructor() {
          super();
          this.db = new Database();
        }
        
        registerRoutes(router) {
          router.get('/analytics/views', this.getViewAnalytics.bind(this));
          router.post('/analytics/track', this.trackView.bind(this));
        }
        
        async getViewAnalytics(req, res) {
          const { documentId } = req.query;
          const views = await this.db.query(
            'SELECT COUNT(*) as views FROM document_views WHERE document_id = ?',
            [documentId]
          );
          res.json(views[0]);
        }
        
        async trackView(req, res) {
          const { documentId, userId } = req.body;
          await this.db.insert('document_views', {
            document_id: documentId,
            user_id: userId,
            viewed_at: new Date()
          });
          res.json({ success: true });
        }
      }
      
      module.exports = AnalyticsExtension;
      ````
    </CodeGroup>
  </Step>
  <Step title="Deploy Extension" icon="upload">
    Upload and activate your extension.
  </Step>
  <Step title="Test Integration" icon="check-circle">
    Verify the extension works correctly.
  </Step>
</Steps>
